name: ⚡ Futuristic README Builder

on:
  workflow_dispatch:
  push:
    branches: [ main ]
    paths:
      - .github/readme.data.yml
      - .github/workflows/readme-futuristic.yml
  schedule:
    - cron: "17 6 * * *" # daily refresh at 06:17 UTC

permissions:
  contents: write

jobs:
  build-readme:
    runs-on: ubuntu-latest
    env:
      EMAIL: ${{ secrets.EMAIL }} # e.g. sathvik9989@gmail.com
      LINKEDIN_URL: ${{ secrets.LINKEDIN_URL }} # e.g. https://www.linkedin.com/in/sathhvik
    steps:
      - name: ⬇️ Checkout
        uses: actions/checkout@v4

      - name: 🔧 Install yq (YAML processor)
        run: |
          sudo wget -qO /usr/local/bin/yq https://github.com/mikefarah/yq/releases/latest/download/yq_linux_amd64
          sudo chmod +x /usr/local/bin/yq
          yq --version

      - name: 📦 Load config
        id: cfg
        run: |
          # Read arrays from .github/readme.data.yml and turn into HTML icon rows
          SKILLS=$(yq -r '.skills[] | "<img src=\"" + .icon + "\" width=\"64\" height=\"64\" title=\"" + .name + "\" style=\"margin:10px;\"/>"' .github/readme.data.yml | paste -sd" " -)
          BADGES=$(yq -r '.badges[] | "  <img src=\"" + .url + "\"/>"' .github/readme.data.yml | paste -sd"\n" -)
          echo "skills_html<<EOF" >> $GITHUB_OUTPUT
          echo "$SKILLS" >> $GITHUB_OUTPUT
          echo "EOF" >> $GITHUB_OUTPUT
          echo "badges_html<<EOF" >> $GITHUB_OUTPUT
          echo "$BADGES" >> $GITHUB_OUTPUT
          echo "EOF" >> $GITHUB_OUTPUT

      - name: 🧪 Sanity check secrets
        run: |
          [ -n "$EMAIL" ] || (echo "Missing secret: EMAIL" && exit 1)
          [ -n "$LINKEDIN_URL" ] || (echo "Missing secret: LINKEDIN_URL" && exit 1)

      - name: 🏗️ Generate README.md
        run: |
          cat > README.md <<'EOF'
<!-- GENERATED BY readme-futuristic.yml — DO NOT EDIT DIRECTLY -->

<h1 align="center">
  <img src="https://readme-typing-svg.herokuapp.com?font=Orbitron&size=32&duration=2600&pause=600&color=00C4FF&center=true&vCenter=true&width=720&lines=👋+Hi%2C+I'm+Sathvik+Reddy;Ops+Manager+%7C+Tech+Strategist+%7C+Cloud+%26+DevOps;Building+systems+that+scale+⚙️" alt="Typing intro"/>
</h1>

<p align="center">
${{ steps.cfg.outputs.badges_html }}
</p>

---

## 🔗 Connect With Me
<p align="center">
  <a href="mailto:${EMAIL}" target="_blank">
    <img src="https://img.icons8.com/fluency/96/gmail-new.png" width="72" height="72" title="Email Me"/>
  </a>
  &nbsp;&nbsp;&nbsp;&nbsp;
  <a href="${LINKEDIN_URL}" target="_blank">
    <img src="https://img.icons8.com/fluency/96/linkedin-circled.png" width="72" height="72" title="LinkedIn Profile"/>
  </a>
</p>

---

## 🛠 Skills & Tools
<p align="center">
  ${{ steps.cfg.outputs.skills_html }}
</p>

---

<p align="center">
  <img src="https://capsule-render.vercel.app/api?type=wave&height=120&color=0:00C4FF,100:0077B5&reversal=true&section=footer"/>
</p>
EOF

      - name: ✅ Link Check (Lychee)
        uses: lycheeverse/lychee-action@v1
        with:
          args: --no-progress --verbose README.md
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: 💾 Commit & Push
        uses: stefanzweifel/git-auto-commit-action@v5
        with:
          commit_message: "chore(readme): auto-generate futuristic README ✨"
          file_pattern: README.md
